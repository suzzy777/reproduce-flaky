+        testNames = new int[tests.size()];
+        testHrefs = new int[tests.size()];
+
+        for (int i = 0; i < tests.size(); i++) {
+            TestRef test = tests.get(i);
+
+            testIds[i] = test.id;
+            testNames[i] = comp.getStringId(test.name);
